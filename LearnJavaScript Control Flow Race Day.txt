let raceNumber = Math.floor(Math.random() * 1000);
const registerEarly = true;
let runnerAge = 18;

if (runnerAge > 18 && registerEarly) {
  raceNumber += 1000;
}
else if (runnerAge > 18 && !registerEarly) {
  console.log(`Your race number is ${raceNumber} and you start at 11:00am.`)
}
else if (runnerAge > 18 && registerEarly) {
  console.log(`Your race number is ${raceNumber} and you start at 9:30am.`);
}
else if (runnerAge < 18) {
  console.log(`Your number is ${raceNumber} and you race at 12:30pm`);
}
else if (runnerAge === 18) {
  console.log('See the registration desk.');
}
